                print([x, y])

//Answer 8
def reverseList(list, size):
    n = len(list)
    start = 0
    while (start<n):
        end = start + size
        tempList = list[start:end]
        tempList.reverse()
        list[start:end] = tempList
        start = start + size
    print(list)

def printNextGreater(list):
    n = len(list)
    for i in range(0, n):
        next = None
        for j in range(i+1, n):
            if list[i] < list[j]:
                next = list[j] 
                break
        print(str(list[i]) + " --> " + str(next))
